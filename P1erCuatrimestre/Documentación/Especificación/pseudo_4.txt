4.1
PROCESO Consulta_Expertos(parámetros_consulta)
  REPETIR
    SI parámetros_consulta = USUARIOS[índice_1] ENTONCES
      MOSTRAR USUARIOS[índice_1]
      REPETIR
        SI EXPERTOS-PROYECTOS[índice_2].id_experto = 
        USUARIOS[índice_1] ENTONCES
          MOSTRAR EXPERTOS-PROYECTOS[índice_2]
          proyecto = EXPERTOS-PROYECTOS[índice_2].id_proyecto
          MOSTRAR PROYECTOS[proyecto].tpo_eval
        FIN SI
      HASTA índice_2 > EXPERTOS-PROYECTOS.número_tuplas
    FIN SI
  HASTA índice_1 > USUARIOS.número_tuplas

4.2
PROCESO Búsqueda_Expertos(párametros_búsqueda)
  CREAR lista_expertos
  REPETIR
    SI parámetros_búsqueda.usuario = USUARIOS[índice] 
    Y USUARIOS[índice].institución DISTINTO 
    PAQUETES_PROYECTOS[párametros_búsqueda.paquete_proyecto].institución
    Y PROYECTOS[párametros_búsqueda.proyecto].área = 
    USUARIOS.[índice].área ENTONCES 
      lista_expertos.AÑADIR(USUARIOS[índice])
    FIN SI
  HASTA índice > USUARIOS.número_tuplas
  DEVOLVER lista_expertos  


4.3 
PROCESO Invitación_Expertos(id_experto)
  REPETIR
    SI USUARIOS[índice].id = id_experto ENTONCES
      correo_invitación <= (PLANTILLAS_CORREOS.asunto = "Invitación") + 
      (PROYECTOS[Elección_1].nombre) + 
      (PAQUETES_PROYECTOS[PROYECTOS[Elección_1].id_paquete) + 
      (USUARIOS[índice].correo)
    FIN SI
  HASTA índice > USUARIOS.número_tuplas


4.4
PROCESO Aceptar_Invitación_Experto(aceptación)
  PROYECTOS[Elección_1].estado <= "En proceso"
  EXPERTOS-PROYECTOS.CREAR_INSTANCIA(aceptación.id_experto,
  aceptación.id_proyecto,(INFORMES_EVALUACIÓN.CREAR_INSTANCIA().id_informe))

4.5
PROCESO Desasignar_Experto/Rechazar_Informe(experto_desasignado)
  informe <= EXPERTOS-PROYECTOS[experto_desasignado.id_experto]
  [experto_desasignado.id_proyecto].id_informe
  BORRAR PROYECTOS[experto_desasignado.id_experto]
  [experto_desasignado.id_proyecto]  
  BORRAR INFORMES_EVALUACIÓN[informe]

4.6 
PROCESO Avisos_De_Plazos(id_experto)
  correo_plazos <= ((PLANTILLAS_CORREOS.asunto = "Aviso Plazos") + 
  PROYECTOS[Elección_1].tpo_inv + 
  EXPERTOS-PROYECTOS[id_expertos][Elección_1].fecha_asignación + 
  PAQUETES_PROYECTOS[Elección2] + USUARIOS[ID_Experto].correo)
  RETORNAR correo_plazos

4.7
PROCESO Valorar_Expertos(valoración_expertos)
  EXPERTOS-PROYECTOS[valoración_expertos.id_experto]
  [valoración_expertos.id_proyecto].valoración = valoración_expertos
